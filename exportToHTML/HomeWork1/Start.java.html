import java.io.*;
import java.util.*;

public class Start {
    static class Point {
        long x, y;

        Point(long x, long y) {
            this.x = x;
            this.y = y;
        }
    }

    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter writer = new PrintWriter(System.out);
        StringTokenizer tokenizer = new StringTokenizer(reader.readLine());

        int n = Integer.parseInt(tokenizer.nextToken());
        int m = Integer.parseInt(tokenizer.nextToken());
        int e = Integer.parseInt(tokenizer.nextToken());

        Point[] basePoints = new Point[n];
        Point[] sourcePoints = new Point[m];
        int[][] segments = new int[e][2];


        for (int i = 0; i < n; i++) {
            tokenizer = new StringTokenizer(reader.readLine());
            long x = Long.parseLong(tokenizer.nextToken());
            long y = Long.parseLong(tokenizer.nextToken());
            basePoints[i] = new Point(x, y);
        }


        for (int i = 0; i < m; i++) {
            tokenizer = new StringTokenizer(reader.readLine());
            long x = Long.parseLong(tokenizer.nextToken());
            long y = Long.parseLong(tokenizer.nextToken());
            sourcePoints[i] = new Point(x, y);
        }


        for (int i = 0; i < e; i++) {
            tokenizer = new StringTokenizer(reader.readLine());
            int a = Integer.parseInt(tokenizer.nextToken()) - 1;
            int b = Integer.parseInt(tokenizer.nextToken()) - 1;
            segments[i][0] = a;
            segments[i][1] = b;
        }

        StringBuilder result = new StringBuilder();

        for (int[] segment : segments) {
            Point a = basePoints[segment[0]];
            Point b = basePoints[segment[1]];
            Point mid = new Point((a.x + b.x) / 2, (a.y + b.y) / 2);

            boolean canDraw = false;
            for (Point source : sourcePoints) {
                if (!intersectsAnySegment(source, mid, basePoints, segments)) {
                    canDraw = true;
                    break;
                }
            }
            result.append(canDraw ? '1' : '0');
        }

        writer.println(result);
        writer.flush();
    }

    private static boolean intersectsAnySegment(Point source, Point mid, Point[] basePoints, int[][] segments) {
        for (int[] segment : segments) {
            Point p1 = basePoints[segment[0]];
            Point p2 = basePoints[segment[1]];
            if (segmentsIntersect(source, mid, p1, p2)) {
                return true;
            }
        }
        return false;
    }

        return false;
    }
}
